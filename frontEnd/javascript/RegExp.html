function findMaxRepeatedChar(str) {
  let maxLength = 0;
  let maxValue = "";
  
  str.replace(/(\w)\1+/g, (match, char) => {
    if (match.length > maxLength) {
      maxLength = match.length;
      maxValue = char;

  });
  
  return { char: maxValue, count: maxLength };
}
